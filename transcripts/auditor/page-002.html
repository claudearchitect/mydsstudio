124	/**
125	 * A prompt-then-read-one-line helper built on raw `line` events instead of
126	 * `readline/promises`'s `rl.question()`. Necessary because `rl.question()`
127	 * has a known gap with piped (non-TTY) stdin: when the whole input is
128	 * available immediately, all `line` events can fire before the first
129	 * `question()` call's internal listener attaches, and the interface's
130	 * `close` event (fired on EOF) never resolves an already-pending
131	 * `question()` promise — the process hangs on "unsettled top-level await."
132	 * Queueing lines as they arrive and resolving a pending waiter (if any)
133	 * handles both piped and interactive stdin correctly.
134	 */
135	function createLineReader(): {
136	  next: (prompt: string) => Promise<string | null>;
137	  close: () => void;
138	} {
139	  const rl = createInterface({ input: stdin, output: stdout });
140	  const queue: string[] = [];
141	  let waiter: ((line: string | null) => void) | null = null;
142	  let closed = false;
143	
144	  rl.on("line", (line: string) => {
145	    if (waiter) {
146	      const w = waiter;
147	      waiter = null;
148	      w(line);
149	    } else {
150	      queue.push(line);
151	    }
152	  });
153	  rl.on("close", () => {
154	    closed = true;
155	    if (waiter) {
156	      const w = waiter;
157	      waiter = null;
158	      w(null);
159	    }
160	  });
161	
162	  return {
163	    next(prompt: string): Promise<string | null> {
164	      stdout.write(prompt);
165	      if (queue.length > 0) {
166	        return Promise.resolve(queue.shift()!);
167	      }
168	      if (closed) {
169	        return Promise.resolve(null);
170	      }
171	      return new Promise((resolve) => {
172	        waiter = resolve;
173	      });
174	    },
175	    close(): void {
176	      rl.close();
177	    },
178	  };
179	}

 * what keeps ids unique across server restarts, page reloads, and concurrent
 * sessions: the client persists belief state in localStorage and replays it
