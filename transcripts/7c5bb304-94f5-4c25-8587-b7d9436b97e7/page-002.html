32	): void {

50	 * malformed JSON, envelope version mismatch, or a BeliefState whose
51	 * `schemaVersion` doesn't match the current contract (z.literal(1)) —
52	 * every failure mode discards silently rather than throwing, per the
53	 * gate's "discard, not crash" requirement.
54	 */

56	  if (typeof window === "undefined") return null;
57	  try {
58	    const raw = window.localStorage.getItem(STORAGE_KEY);
59	    if (!raw) return null;
60	    const parsed = JSON.parse(raw) as Partial<PersistedSession>;
61	    if (parsed.envelopeVersion !== ENVELOPE_VERSION) return null;
62	
63	    const beliefStateResult = BeliefStateSchema.safeParse(parsed.beliefState);
64	    if (!beliefStateResult.success) return null;
65	    // Redundant with the schema (schemaVersion is z.literal(1)) but kept
66	    // explicit since it's the exact clause the gate calls out.
67	    if (beliefStateResult.data.schemaVersion !== 1) return null;
68	
69	    const transcript = Array.isArray(parsed.transcript) ? parsed.transcript : [];
70	
71	    return {
72	      envelopeVersion: ENVELOPE_VERSION,
73	      beliefState: beliefStateResult.data,
74	      transcript: transcript as TranscriptEntry[],
75	      savedAt: typeof parsed.savedAt === "string" ? parsed.savedAt : new Date().toISOString(),
76	    };
77	  } catch {
78	    return null;
79	  }
80	}
